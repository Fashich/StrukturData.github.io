Selection Sort

 using System;
                using System.Collections.Generic;
                using System.Linq;
                using System.Text;
                using System.Threading.Tasks;

                namespace _2._2_SelectionSort_Langkah_Rincian
                {
                    internal class Program
                    {
                        static void Selectionsort(int[] array)
                        {
                            int nilai = array.Length;
                            for (int iterasi = 0; iterasi < nilai - 1; iterasi++)
                            {
                                Console.WriteLine("\niterasi {0}.", iterasi);
                                int maximalindex = 0;
                                for (int i = 1; i < nilai - iterasi; i++)
                                {
                                    if (array[iterasi] < array[maximalindex])
                                    {
                                        maximalindex = i;
                                    }
                                }
                                int tempatbariselemen = array[maximalindex];
                                array[maximalindex] = tempatbariselemen;
                                Console.WriteLine("Array setelah iterasi {0}.", iterasi);
                                MenampilkanArray(array);
                            }
                        }
                        static void MenampilkanArray(int[] array)
                        {
                            foreach (int angka in array)
                            {
                                Console.WriteLine(angka + " ");
                            }
                            Console.WriteLine();
                        }
                        static void Main(string[] args)
                        {
                            int[] array = { 2, 4, 6, 8, 10, 12, 14, 16 };
                            Console.WriteLine("Array sebelum pengurutan");
                            MenampilkanArray(array);
                            Selectionsort(array);
                            Console.WriteLine("\nArray setelah pengurutan");
                            MenampilkanArray(array);
                        }
                    }
                }
