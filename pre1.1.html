Mengilustrasikan Bubble Sort dan Selection Sort

Bubble Sort. Ini adalah salah satu algoritma pengurutan sederhana yang bekerja dengan cara membandingkan dan menukar elemen-elemen yang berdekatan jika mereka tidak berada dalam urutan yang benar. Proses ini berulang kali diulangi hingga tidak ada lagi pertukaran yang perlu dilakukan, menandakan bahwa array sudah dalam urutan yang benar. Misalkan kita memiliki array [5, 3, 8, 4, 2]. Pada langkah pertama, Bubble Sort akan membandingkan elemen pertama (5) dengan elemen kedua (3). Karena 5 lebih besar dari 3, pertukaran akan dilakukan, dan array akan menjadi [3, 5, 8, 4, 2]. Proses ini akan terus berlanjut ke sepanjang array, memindahkan elemen terbesar ke posisi terakhir. Setelah iterasi pertama selesai, elemen terbesar akan berada di posisi terakhir. Selanjutnya, algoritma akan kembali melakukan iterasi pada array yang lebih pendek (tanpa elemen terakhir yang sudah terurut). Proses ini akan terus berlanjut sampai tidak ada lagi pertukaran yang dibutuhkan. Selanjutnya, mari kita bahas Selection Sort. Algoritma ini bekerja dengan cara memilih elemen terkecil dari array dan menukarnya dengan elemen pertama. Kemudian, algoritma akan memilih elemen terkecil dari sisa array (tanpa elemen pertama yang sudah terurut) dan menukarnya dengan elemen kedua, dan seterusnya, hingga array terurut sepenuhnya. Menggunakan contoh yang sama dengan sebelumnya ([5, 3, 8, 4, 2]), pada langkah pertama Selection Sort akan mencari elemen terkecil dari seluruh array (2), lalu menukarnya dengan elemen pertama. Array akan menjadi [2, 3, 8, 4, 5]. Kemudian, algoritma akan mencari elemen terkecil dari sisa array (3, 8, 4, 5) dan menukarnya dengan elemen kedua, dan seterusnya. Proses ini akan terus berlanjut hingga seluruh array terurut secara ascending.

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace _1._1___bubble___selection_sort
{
    internal class Program
    { 
        static void Bubblesort(int[] array)
        {
            int nilai = array.Length; //semua baris elemen
            for (int i = 0; i < nilai - 1; i++)
            {
                for (int j = 0; j < nilai - i - 1; j++)
                {
                    if (array[j] > array[j + 1])
                    {
                        int tempatbariselemen = array[j];
                        array[j] = array[j + 1];
                        array[j + 1] = tempatbariselemen;
                    }
                }
            }
        }
        static void Selectionsort(int[] array)
        {
            int nilai = array.Length;
            for (int i = 0; i < nilai - 1;  i++)
            {
                int minimalindex = i;
                for (int j = i + 1; j < nilai; j++)
                {
                    if (array[j] < array[minimalindex])
                    {
                        minimalindex = j;
                    }
                }
                int tempatbariselemen = array[minimalindex];
                array[minimalindex] = array[i];
                array[i] = tempatbariselemen;
            }
        }
        // menampilkan array
        static void menampilkanarray(int[] array)
        {
            foreach (int angka in array)
            {
                Console.WriteLine(angka + " ");
            }
            Console.WriteLine();
        }
        static void Main(string[] args)
        {
            int[] arraybubble = { 360, 140, 280, 720, 480, 1440, 1080, 4000, 2160, 16000, 8000 };
            int[] arrayselection = { 100000, 75000, 50000, 20000, 10000, 5000, 2000, 1000, 500, 200, 100, 50, 1 };
            Console.WriteLine("array sebelum Bubble Sort: ");
            menampilkanarray(arraybubble);
            Bubblesort(arraybubble);
            Console.WriteLine("array setelah Bubble Sort: ");
            menampilkanarray(arraybubble);
            Console.WriteLine("\narray sebelum selection sort");
            menampilkanarray(arrayselection);
            Selectionsort(arrayselection);
            Console.WriteLine("Array setelah selection sort: ");
            menampilkanarray(arrayselection);
        }
    }
}
